         def validates_uniqueness_of(*attr_names)
           attr_names.each do |attr|
             add_validation_to_chain do |record|
               value = record.send(attr)               
               if self.send("find_by_#{attr.to_s}",value) 
                 record.errors.add(attr.to_s,"must be unique")
               end
             end
           end           
         end
