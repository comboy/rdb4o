         def method_missing(name, *args)
            if match = /^find_by_([_a-zA-Z]\w*)$/.match(name.to_s)
               attribute_name = match[1]
               class_proto = self.new
               class_proto.method("#{attribute_name}=").call args[0]
               result = self.database.get class_proto
               if result.empty?
                 nil
               else
                 result[0]
               end
            end
         end
